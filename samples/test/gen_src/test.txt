component PositionComponent {
	members {
		position : Halley::Vector2f
	}
}

component VelocityComponent {
	members {
		velocity : Halley::Vector2f
	}
}

component TimeComponent {
	members {
		elapsed : float
	}
}

component SpriteComponent {
	members {
		sprite : Halley::Sprite
		layer : int
	}
}

component SpriteAnimationComponent {
	members {
		player : Halley::AnimationPlayer
	}
}

system TimeSystem {
	family main {
		TimeComponent
	}

	strategy = individual
	access = pure
	language = C++
	subDivide = 1
}

system MovementSystem {
	family main {
		PositionComponent
		VelocityComponent
	}

	strategy = individual
	access = pure
	language = C++
	subDivide = 1
}

system SpriteAnimationSystem {
	family main {
		SpriteComponent
		SpriteAnimationComponent
		const PositionComponent
		const VelocityComponent
	}

	strategy = individual
	access = pure
	language = C++
	subDivide = 1
}

system RenderSystem {
	family main {
		PositionComponent
		SpriteComponent
	}

	method = render
	strategy = default
	access = pure
	language = C++
	subDivide = 1
}
